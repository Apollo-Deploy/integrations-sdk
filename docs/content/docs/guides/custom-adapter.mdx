You can build adapters for providers not included in the SDK using `defineAdapter()`. The factory enforces the full `IntegrationAdapter` contract at compile time.

## Scaffold your package

```typescript title="packages/adapter-notion/src/adapter.ts" lineNumbers
import {
  defineAdapter,
  type AdapterDefinition,
} from '@apollo-deploy/integrations';
import { createNotionOAuth } from './oauth.js';
import { createNotionWebhook } from './webhook.js';
import { createNotionIssueTracking } from './capabilities/issue-tracking.js';
import type { NotionAdapterConfig } from './types.js';

export const createNotionAdapter = defineAdapter<NotionAdapterConfig>({
  id: 'notion',
  name: 'Notion',
  metadata: {
    description: 'Connect Notion databases as a lightweight issue tracker.',
    category: 'Project Management',
    dateAdded: '2024-09-01',
    websiteUrl: 'https://notion.so',
    docsUrl: 'https://developers.notion.com',
  },
  capabilities: ['issue-tracking'] as const,

  tokenMetadata: {
    expiresInSeconds: null, // Notion tokens do not expire
    refreshable: false,
    rotatesRefreshToken: false,
    requiresRefreshLock: false,
  },

  createOAuthHandler: (config) => createNotionOAuth(config),
  createWebhookHandler: (config) => createNotionWebhook(config),
  createIssueTracking: (config) => createNotionIssueTracking(config),
});
```

## Implement OAuthHandler

```typescript title="packages/adapter-notion/src/oauth.ts" lineNumbers
import type { OAuthHandler } from '@apollo-deploy/integrations';
import type { NotionAdapterConfig } from './types.js';

export function createNotionOAuth(config: NotionAdapterConfig): OAuthHandler {
  return {
    getAuthorizationUrl({ state, redirectUri }) {
      const params = new URLSearchParams({
        client_id: config.clientId,
        redirect_uri: redirectUri,
        response_type: 'code',
        state,
        owner: 'user',
      });
      return `https://api.notion.com/v1/oauth/authorize?${params}`;
    },

    async exchangeCode({ code, redirectUri }) {
      const res = await fetch('https://api.notion.com/v1/oauth/token', {
        method: 'POST',
        headers: {
          Authorization: `Basic ${Buffer.from(`${config.clientId}:${config.clientSecret}`).toString('base64')}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ grant_type: 'authorization_code', code, redirect_uri: redirectUri }),
      });
      const data = await res.json();
      return {
        accessToken: data.access_token,
        scope: 'read write',
        providerData: {
          workspaceId: data.workspace_id,
          workspaceName: data.workspace_name,
          botId: data.bot_id,
        },
      };
    },

    async refreshToken() {
      throw new Error('Notion tokens do not expire and cannot be refreshed.');
    },

    async getIdentity(accessToken) {
      const res = await fetch('https://api.notion.com/v1/users/me', {
        headers: { Authorization: `Bearer ${accessToken}`, 'Notion-Version': '2022-06-28' },
      });
      const user = await res.json();
      return {
        providerAccountId: user.id,
        displayName: user.name,
        avatarUrl: user.avatar_url,
        metadata: {},
      };
    },
  };
}
```

## Implement WebhookHandler

```typescript title="packages/adapter-notion/src/webhook.ts" lineNumbers
import { createHmac, timingSafeEqual } from 'node:crypto';
import type { WebhookHandler, IntegrationEvent } from '@apollo-deploy/integrations';
import type { NotionAdapterConfig } from './types.js';

export function createNotionWebhook(_config: NotionAdapterConfig): WebhookHandler {
  return {
    supportedEvents: ['page.created', 'page.updated', 'database.updated'],

    verifySignature({ rawBody, headers, secret }) {
      const sig = headers['notion-signature'];
      if (!sig) return false;
      const expected = createHmac('sha256', secret).update(rawBody).digest('hex');
      return timingSafeEqual(Buffer.from(sig), Buffer.from(expected));
    },

    parseEvent({ body, headers }): IntegrationEvent {
      const payload = body as Record<string, unknown>;
      return {
        provider: 'notion',
        eventType: String(payload.type),
        domain: 'issue-tracking',
        deliveryId: String(payload.id),
        correlationId: String(payload.id),
        timestamp: new Date(),
        data: payload,
        rawPayload: body,
      };
    },

    getDeliveryId(headers) {
      return headers['notion-webhook-id'] ?? 'unknown';
    },
  };
}
```

## Add AdapterMetadata

The `metadata` field is optional but strongly recommended — it powers `hub.listAdapters()`:

```typescript
metadata: {
  iconUrl: 'https://www.notion.so/images/favicon.ico',
  description: 'Connect Notion databases as a lightweight issue tracker.',
  category: 'Project Management',
  dateAdded: '2024-09-01',
  websiteUrl: 'https://notion.so',
  docsUrl: 'https://developers.notion.com',
},
```

## Register the adapter

```typescript title="lib/integrations.ts" lineNumbers
import { createNotionAdapter } from '@packages/adapter-notion';

const hub = new IntegrationHub({
  adapters: {
    notion: createNotionAdapter({
      clientId: process.env.NOTION_CLIENT_ID!,
      clientSecret: process.env.NOTION_CLIENT_SECRET!,
    }),
  },
  crypto: createCryptoProviderFromEnv(),
  logger: console,
});
```
